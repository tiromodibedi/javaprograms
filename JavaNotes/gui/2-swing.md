## A better way of Defining a JFrame
* Make it a derived class of the parent class JFrame

```java
import javax.swing.JFrame;
import javax.swing.JButton;

public class SecondSwingDemo extends JFrame{
  public static final int WIDTH = 300;
  public static final int HEIGHT = 200;

  public SecondSwingDemo(){
    super();
    setSize(WIDTH, HEIGHT);
    setTitle("A better Implementation of a Windowin system");
    setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);

    JButton endButton = new JButton("Click to end program");
    endButton.addActionListener(new EndingListener());
    add(endButton);
  }
}
```
* Now to Demo the code above
```java
public class DemoWindow{
  public static void main(String[] args) {
    SecondSwingDemo w = new SecondSwingDemo();
    w.setVisible(true);
  }
}
```

```java
import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class WordSearch extends JFrame implements ActionListener{

    private JButton exitButton,b;
    private Container pane;
    private JLabel l;
    private JTextField text;
    private JPanel canvas;
    private Container buttons;

    public  void actionPerformed(ActionEvent e) {
	System.out.println(e);
    }

    public  WordSearch() {
	this.setTitle("Can I?\nGuess not");
	this.setSize(800,600);
	this.setLocation(200,50);
	this.setDefaultCloseOperation(EXIT_ON_CLOSE);

	pane = this.getContentPane();
	//pane.setLayout(new FlowLayout());
	pane.setLayout(new GridLayout(6,2));

	exitButton = new JButton("Exit");
	b = new JButton("Button 1");
	l = new JLabel("This is a label",null,JLabel.CENTER);
	text = new JTextField(40);
	canvas = new JPanel();
	canvas.setBorder(BorderFactory.createLineBorder(Color.blue,2));

	buttons = new Container();
	//buttons.setLayout(new FlowLayout());
	buttons.setLayout(new GridLayout(0,2));
	buttons.add(b);
	buttons.add(exitButton);
	pane.add(buttons);
	pane.add(l);
	pane.add(text);
	pane.add(canvas);

	exitButton.addActionListener(this);
	b.addActionListener(this);
    }



    public  void hello() {
	System.out.println("Hello");
    }
    public static  void main(String[] args) {
	WordSearch g = new WordSearch();
	g.setVisible(true);
    }
}
```
